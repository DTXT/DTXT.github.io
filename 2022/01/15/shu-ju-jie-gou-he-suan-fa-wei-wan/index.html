二分法查找

package 线性结构.数组;

/**
 * @author: lpc
 * @date: 31/1/2022 下午 5:23
 * @describe: 二分法查找:不断查找中间位置的值
 * @vision
 */
public class BinarySearch {
   public static void main(String[] args) {
      int[] arr = {1, 2, 3, 4, 5, 6, 7, 8, 9};
      int target = 19;
      int i = search(arr, target);
      System.out.println(target + "在数组的" + i + "索引！");
   }

   public static int search(int[] arr, int target) {
      // 开始位置
      int begin = 0;
      // 结束位置
      int end = arr.length - 1;
      // 中间位置
      int mid = (begin + end) / 2;
      while (true) {
         // 4、结束条件：数组里没有该元素
         if (begin >= end) {
            return -1;
         }
         // 1、判断中间元素之是不是目标
         if (arr[mid] == target) {
            return mid;
         } else {
            // 2、不是
            // 2.1、中间元素大于target，则它就在数组前边
            if (arr[mid] > target) {
               // 调整end，结束位置
               end = mid - 1;
            } else {
               // 2.2、 小于target，则目标元素在数组后边
               // 调整begin，开始位置
               begin = mid + 1;
            }
            // 3、调整中间值
            mid = (begin + end) / 2;
         }
      }
   }
}

package 线性结构.栈;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 31/1/2022 下午 5:51
 * @describe: 模拟栈结构：先进后出
 * @vision
 */
public class Stack {
  public static void main(String[] args) {
    //栈
    Stack stack = new Stack();
    // 压栈
    for (int i = 0; i < 5; i++) {
      stack.push(i);
    }
    // 栈的状态
    System.out.println(Arrays.toString(stack.status()));
    // 出栈
    int pop = stack.pop();
    System.out.println("取出栈顶元素："+pop);
    // 栈的状态
    System.out.println(Arrays.toString(stack.status()));
  }

  // 栈的底层使用数组来存储
  int[] elements;

  public Stack() {
    elements = new int[0];
  }
  // 压栈
  public void push(int element) {
    // 创建新数组
    int[] array = new int[elements.length + 1];
    // 复制原来的数组
    for (int i = 0; i < elements.length; i++) {
      array[i] = elements[i];
    }
    // 添加新加入的元素
    array[elements.length] = element;
    // 替换旧数组
    elements = array;
  }
  // 出栈
  public  int pop() {
    if(elements.length==0){
      throw new RuntimeException("stack is empty");
    }
    // 取出最后一个元组(栈顶元素)
    int element = elements[elements.length - 1];
    // 创建新数组
    int[] array = new int[elements.length - 1];
    // 复制元素组，除了最后一个
    for (int i = 0; i < elements.length - 1; i++) {
      array[i] = elements[i];
    }
    // 替换旧数组
    elements = array;
    // 返回
    return element;
  }
  // 查看栈的全部元素
  public int[] status() {
    return elements;
  }
}

队列结构

package 线性结构.队列;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 31/1/2022 下午 6:19
 * @describe: 队列
 * @vision
 */
public class Queue {
  public static void main(String[] args) {
    // 队列
    Queue queue = new Queue();
    // 入队
    for (int i = 0; i < 5; i++) {
      queue.add(i);
    }
    System.out.println(Arrays.toString(queue.status()));
    // 出队
    int poll = queue.poll();
    System.out.println("取出头元素"+poll);
    System.out.println(Arrays.toString(queue.status()));
  }
  // 队列，使用数组来存储
  int[] elements;

  public Queue() {
    elements = new int[0];
  }
  // 入队 ，和栈的入栈一样
  public void add(int element) {
    // 创建新数组
    int[] array = new int[elements.length + 1];
    // 复制原来的数组
    for (int i = 0; i < elements.length; i++) {
      array[i] = elements[i];
    }
    // 添加新加入的元素
    array[elements.length] = element;
    // 替换旧数组
    elements = array;
  }
  // 出队，返回队列头部数据(和栈相反)
  public int poll() {
    if (elements.length == 0) {
      throw new RuntimeException("queue is empty");
    }
    // 头元素
    int element = elements[0];
    // 其他元素在组成一个队列
    int[] array = new int[elements.length - 1];
    for (int i = 0; i < array.length; i++) {
      array[i] = elements[i + 1];
    }
    // 替换旧数组
    elements = array;
    return element;
  }
  // 查看栈的全部元素
  public int[] status() {
    return elements;
  }
}

单链表

一个节点连接着一个节点，每个节点的存储位置不同

节点

package 线性结构.单链表;

/**
 * @author: lpc
 * @date: 31/1/2022 下午 6:55
 * @describe: 节点:节点内容 下一个节点
 * @vision
 */
public class Node {
  // 节点内容
  int data;
  // 下一个节点
  Node next;

  public Node(int data) {
    this.data = data;
  }
  // 追加节点,找到最后的一个节点，添加，再返回添加的这个节点
  public Node append(Node node) {
    if (next == null) {
      this.next = node;
      return node;
    } else {
      this.next.append(node);
    }
    return null;
  }
  // 获取下一个节点
  public Node next() {
    return this.next;
  }
  // 获取当前节点内容
  public int getData() {
    return this.data;
  }
  // 删除这个节点的下一个节点,并返回被删除的节点
  public Node removeNext() {
    Node next1 = this.next;
    Node next2 = this.next.next;
    this.next = next2;
    return next1;
  }
  // 显示所有节点
  public void show() {
    if (this.data != 0) {
      System.out.print(this.data + " ");
    }
    if (this.next != null) {
      this.next.show();
    }
  }
  // 在这个节点后边插入一个节点
  public void insertNode(Node node) {
    Node next = this.next;
    this.next = node;
    node.next = next;
  }

  @Override
  public String toString() {
    return "Node{" + "data=" + data + ", next=" + next + '}';
  }
}

双向循环链表

节点:指向上一个节点和下一个节点

package 线性结构.双向循环链表;

import com.sun.org.apache.bcel.internal.generic.RET;

/**
 * @author: lpc
 * @date: 1/2/2022 上午 9:32
 * @describe:
 * @vision
 */
public class DoubleNode {
  // 上一个节点
  DoubleNode pre=this;
  // 下一个节点
  DoubleNode next=this;
  // 节点数据
  int data;

  public DoubleNode(int data) {
    this.data = data;
  }
  //  增加节点
  public void after(DoubleNode node) {
    // 原来的下一个节点
    DoubleNode next = this.next;
    // 现在的下一个节点
    this.next = node;
    // 这个新节点的前一个节点
    node.pre = this;
    // 这个新节点的下一个节点
    node.next = next;
    // 这个新节点的下一个节点的前一个节点
    next.pre = node;
    // 需要修改的有三个节点：这个节点(next属性) (pre属性)新节点(next属性) (pre属性)这个节点的下一个节点
    // 这个节点的前一个节点会自动改变，新节点的下一个节点设置为next,next=this 连带this.pre也修改了值为新节点
  }
  // 下一个节点
  public DoubleNode next() {
    return this.next;
  }
  // 上一个节点
  public DoubleNode pre() {
    return this.pre;
  }
  // 获取节点数据
  public int getData() {
    return this.data;
  }
}

    // 第一项或第二项的值为1
    if (n == 1 || n == 2) {

    //这一项的前两项的和
    return show(n - 1) + show(n - 2);
  }
}

package 排序算法.交换排序;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 上午 11:13
 * @describe:
 * @vision
 */
public class 冒泡排序 {
  public static void main(String[] args) {
    //
    int[] arr = {1, 3, 5, 2, 4, 99, 55, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr);
    System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    // 比较长度-1轮
    for (int i = 0; i < arr.length - 1; i++) {
      // 每轮比较 长度-1-确定的位数
      // 第一轮比较 长度-1，后面每次-1，因为每次一轮比较后确定一位最大数
      for (int j = 0; j < arr.length - 1 - i; j++) {
        if (arr[j] > arr[j + 1]) {
          int temp = arr[j];
          arr[j] = arr[j + 1];
          arr[j + 1] = temp;
        }
      }
    }
  }
}

2、快速排序

package 排序算法.交换排序;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 上午 11:23
 * @describe:
 * @vision
 */
public class 快速排序 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr, 0, arr.length - 1);
    System.out.println(Arrays.toString(arr));
  }

  /**
   * 从开始start排到结尾end
   *
   * @param arr
   * @param start
   * @param end
   */
  public static void show(int[] arr, int start, int end) {
    if (start < end) {
      // 把数组中的第0个数字作为标准数
      int stard = arr[start];
      // 记录需要排序的下标
      int low = start;
      int high = end;
      // 循环找比标准数大的数和小的数，进行分类
      while (low < high) {
        // 右边数字比标准数大，下标向前，移动下标
        while (low < high && stard <= arr[high]) {
          high--;
        }
        // 右边数字比标准数小，使用右边的数字替换左边的数字
        arr[low] = arr[high];
        // 左边数字倍比标准数小，就移动下标
        while (low < high && arr[low] <= stard) {
          low++;
        }
        //替换
        arr[high] = arr[low];
      }
      // 标准数在中间
      arr[low] = stard;
      // 处理左边
      show(arr, start, low);
      // 处理右边
      show(arr, low + 1, end);
    }
  }
}

3、直接插入排序

package 排序算法.插入排序;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 下午 1:26
 * @describe: 确定左边的数，左边的都是小的，从右边拿一个依次和前边比
 * @vision
 */
public class 直接插入排序 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr);
    System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    // 遍历
    for (int i = 1; i < arr.length; i++) {
      if (arr[i - 1] > arr[i]) {
        int temp = arr[i];
        int j;
        // 遍历当前数字的前方的数字
        for (j = i - 1; j >= 0 && arr[j] > temp; j--) {
          //给后面的赋值
          arr[j+1] = arr[j];
        }
        //给前面的赋值。注意j--
        arr[j + 1] = temp;
      }
    }
  }
}

4、希尔排序

package 排序算法.插入排序;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 下午 3:23
 * @describe:
 * @vision
 */
public class 希尔排序 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr);
    System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    // 遍历步长
    for (int d = arr.length / 2; d > 0; d /= 2) {
      // 遍历d后边所有元素
      for (int i = d; i < arr.length; i++) {
        // 遍历d前边的所有元素
        for (int j = i - d; j >= 0; j -= d) {
          if (arr[j] > arr[j + d]) {
            int temp = arr[j];
            arr[j] = arr[j + d];
            arr[j + d] = temp;
          }
        }
      }
    }
  }
}

package 排序算法.选择排序;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 下午 4:20
 * @describe:
 * @vision
 */
public class 简单选择排序 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr);
    System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    for (int i = 0; i < arr.length; i++) {
      int index = i;
      for (int j = i + 1; j < arr.length; j++) {
        if (arr[index] > arr[j]) {
          index = j;
        }
      }
      //本身就是最小的，就不需要交换了
      if (index != i) {
        int temp = arr[index];
        arr[index] = arr[i];
        arr[i] = temp;
      }
    }
  }
}

package 排序算法;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 下午 7:37
 * @describe:
 * @vision
 */
public class 基数排序 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
    System.out.println(Arrays.toString(arr));
    show(arr);
    System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    // 找出最大的数字存数组中
    int max = Integer.MIN_VALUE;
    for (int i = 0; i < arr.length; i++) {
      if (arr[i] > max) {
        max = arr[i];
      }
    }
    // 最大数是几位数
    int maxLength = (max + "").length();
    // 创建二维数组，存储数据(例:个位是1，存1号数组，百位是2存2号数组)
    int[][] temp = new int[10][arr.length];
    // 临时数组，记录二维数组中每位存入数字的个数
    int[] counts = new int[10];
    // 根据几位数确定比较次数(99是两位数，就比较2次,比较个位和十位)
    for (int i = 0, n = 1; i < maxLength; i++, n *= 10) {
      // 遍历每一个数字，
      for (int j = 0; j < arr.length; j++) {
        // 取出每一位的数字(关键)
        int ys = arr[j] / n % 10;
        // 当前数字放入二维数组中指定的位置
        temp[ys][counts[ys]] = arr[j];
        // 个数++
        counts[ys]++;
      }
      // 取出数字
      int w = 0;
      for (int j = 0; j < counts.length; j++) {
        if (counts[j] != 0) {
          // 遍历个数
          for (int k = 0; k < counts[j]; k++) {
            // 取出元素,放回arr
            arr[w] = temp[j][k];
            w++;
          }
          // 记录个数置为0，下次要用
          counts[j] = 0;
        }
      }
    }
  }
}

优化加入队列

package 排序算法;

import 线性结构.队列.Queue;

import java.util.Arrays;

/**
 * @author: lpc
 * @date: 1/2/2022 下午 7:37
 * @describe:
 * @vision
 */
public class 基数排序加入队列 {
  public static void main(String[] args) {
    int[] arr = {55, 1, 3, 5, 2, 4, 99, 66, 77, 44, 33};
        System.out.println(Arrays.toString(arr));
    show(arr);
        System.out.println(Arrays.toString(arr));
  }

  public static void show(int[] arr) {
    // 找出最大的数字存数组中
    int max = Integer.MIN_VALUE;
    for (int i = 0; i < arr.length; i++) {
      if (arr[i] > max) {
        max = arr[i];
      }
    }
    // 最大数是几位数
    int maxLength = (max + "").length();
    // 创建10个队列
    Queue[] temp = new Queue[10];
    // 为队列数组赋值
    for (int i = 0; i < temp.length; i++) {
      temp[i] = new Queue();
    }
    // 根据几位数确定比较次数(99是两位数，就比较2次,比较个位和十位)
    for (int i = 0, n = 1; i < maxLength; i++, n *= 10) {
      // 遍历每一个数字，
      for (int j = 0; j < arr.length; j++) {
        // 取出每一位的数字(关键)
        int ys = arr[j] / n % 10;
        // 当前数字放入指定的队列
        temp[ys].add(arr[j]);
      }
      // 取出数字
      int w = 0;
      for (int j = 0; j < temp.length; j++) {
        // 遍历每个队列的所有元素
        while (temp[j].status().length != 0) {
          // 取出元素,放回arr
          arr[w] = temp[j].poll();
          w++;
        }
      }
    }
  }
}

    // 创建树
    BinaryTree tree = new BinaryTree();
    // 创建根节点
    TreeNode root = new TreeNode(1);
    // 把根节点赋给树
    tree.setRoot(root);
    // 创建两个节点
    TreeNode rootL = new TreeNode(2);
    TreeNode rootR = new TreeNode(3);
    // 设置为根节点的子节点
    root.setLeftNode(rootL);
    root.setRightNode(rootR);
    // 创建子节点，给子节点添加子节点
    rootL.setLeftNode(new TreeNode(4)); //          1
    rootL.setRightNode(new TreeNode(5)); //     2        3
    rootR.setLeftNode(new TreeNode(6)); //    4   5    6   7
    rootR.setRightNode(new TreeNode(7));
    //  前序遍历 根左右
    System.out.println("前中后遍历");
    tree.fronShow();
    System.out.println();
    //  中序遍历 左根右
    tree.midShow();
    System.out.println();
    //  后序遍历 左右根
    tree.afterShow();
    System.out.println();
    System.out.println("前序查找");
    TreeNode result = tree.frontSearch(1);
    System.out.println(result);
    // 删除一个节点
    System.out.println("删除一个节点");
    tree.fronShow();
    tree.delete(2);
    System.out.println();
    tree.fronShow();
  }

// 节点
class TreeNode {
  // 权
  int value;
  // 左节点
  TreeNode leftNode;
  // 右节点
  TreeNode rightNode;

  public TreeNode(int value) {
    this.value = value;
  }
  // 设置左节点
  public void setLeftNode(TreeNode leftNode) {
    this.leftNode = leftNode;
  }
  // 设置右节点
  public void setRightNode(TreeNode rightNode) {
    this.rightNode = rightNode;
  }
  // 前序遍历 根左右
  public void fronShow() {
    // 遍历字自己
    System.out.print(value + " ");
    // 左
    if (leftNode != null) {
      leftNode.fronShow();
    }
    // 右
    if (rightNode != null) {
      rightNode.fronShow();
    }
  }
  // 中序遍历 左根右
  public void midShow() {
    // 左
    if (leftNode != null) {
      leftNode.midShow();
    }
    // 遍历字自己
    System.out.print(value + " ");
    // 右
    if (rightNode != null) {
      rightNode.midShow();
    }
  }
  // 后序遍历 左右根
  public void afterShow() {
    // 左
    if (leftNode != null) {
      leftNode.afterShow();
    }
    // 右
    if (rightNode != null) {
      rightNode.afterShow();
    }
    // 遍历字自己
    System.out.print(value + " ");
  }
  // 前序查找
  public TreeNode frontSearch(int i) {
    TreeNode target = null;
    // 对比当前节点
    if (this.value == i) {
      return this;
    } else {
      // 对比左节点
      if (leftNode != null) {
        target = leftNode.frontSearch(i);
      }
      if (target != null) {
        return target;
      }
      // 对比右节点
      if (rightNode != null) {
        target = rightNode.frontSearch(i);
      }
      return target;
    }
  }
  // 删除一个节点
  public void delete(int i) {
    TreeNode parent = this;
    // 左边
    if (parent.leftNode != null && parent.leftNode.value == i) {
      parent.leftNode = null;
      return;
    }
    // 右边
    if (parent.rightNode != null && parent.rightNode.value == i) {
      parent.rightNode = null;
      return;
    }
    // 没有，就去子节点找
    parent = leftNode;
    if (parent != null) {
      parent.delete(i);
    }
    parent = rightNode;
    if (parent != null) {
      parent.delete(i);
    }
  }

  @Override
  public String toString() {
    return "TreeNode{"
        + "value="
        + value
        + ", leftNode="
        + leftNode
        + ", rightNode="
        + rightNode
        + '}';
  }
}
